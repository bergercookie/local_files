function end_sim = DRprogressbar(simtime,tf,h,Ts)

% PROGRESSBAR plots a progress bar while the simulation is being evaluated.
%   This function is called by 'progressbar' block.
%
%  syntax:
%        end_sim = DRprogressbar(simtime,tf,h,Ts)
%
% Copyright (C) 2009, by Carmine Dario Bellicoso and Marco Caputano.
% This file is part of GNU LGPLv2.1 DAMA^{ROB}
%

end_sim = 0;

% Check if the progress bar has been closed
if ishandle(h)
    % If tf still hasn't been initialized in Simulink, set the progress bar to zero
    if tf == 0
        waitbar(0,h);
        % Update the progressbar every 100*Ts sample, where Ts is the sampling time
    elseif mod(simtime,Ts) == 0
        waitbar(simtime/tf,h,['Simulating system...   ' ...
            num2str(round(simtime*100/tf)) '%']);
        set(h,'Visible','on');
    end
end
if simtime == tf
    % close(h);
end
end
