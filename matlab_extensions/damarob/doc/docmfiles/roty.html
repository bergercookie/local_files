function R = roty(beta)

% ROTY generates rotation matrix (direction cosine matrix) for a Y-axis
% rotation.
%  syntax: R = roty(beta)
%
%  where:
%       beta --> rotation angles (measured in radians)
%
% Copyright (C) 2009, by Carmine Dario Bellicoso and Marco Caputano.
% This file is part of GNU LGPLv2.1 DAMA^{ROB}
%    http://www.damarob.altervista.org
%

R = [cos(beta)   0   sin(beta);
     0           1   0;
     -sin(beta)  0   cos(beta)];

end
